JavaScript 문법 요약 카드

(), [], {} 차이

  • () : 함수 호출, 우선순위 지정
  • [] : 배열 만들기, 인덱스로 접근
  • {} : 코드 블록, 객체 생성
function greet() { return "hello"; }      // {}: 함수 블록
let arr = [1, 2, 3];                      // []: 배열
let person = { name: "yeon" };            // {}: 객체

변수 선언: let, const, var

let a = 10;     // 블록 범위, 재할당 가능
const b = 20;   // 블록 범위, 재할당 불가
var c = 30;     // 함수 범위, 재선언 가능 (요즘 잘 안 씀)
a = 10, b = 20, c = 30

배열 (Array)

let fruits = ["🍎", "🍌", "🍇"];
console.log(fruits[1]);
배열은 []로 만들고, 인덱스로 접근해요.
🍌

조건문: if / else

let isLoggedIn = true;

if (isLoggedIn) {
console.log("hello!");
} else {
console.log("bye");
}
hello!

let isLoggedIn = false;

if (isLoggedIn) {
console.log("hello!");
} else {
console.log("bye");
}
bye

반복문: for / for...of / forEach / map

let fruits = ["🍎", "🍌", "🍇"];

// for: 인덱스를 이용해서 배열을 하나씩 출력
for (let i = 0; i < fruits.length; i++) {
console.log(fruits[i]);
}
🍎 🍌 🍇

// for...of: 배열 값을 하나씩 바로 꺼내서 출력
for (let fruit of fruits) {
console.log(fruit);
}
🍎 🍌 🍇

// forEach(): 배열의 각각에 대해 함수를 실행
fruits.forEach(function(fruit) {
console.log(fruit);
});
0: 🍎
1: 🍌
2: 🍇

//map(): 배열의 값을 바꿔서 새로운 배열을 만들기
let upperFruits = fruits.map(fruit => fruit.toUpperCase());
console.log(upperFruits);
"🍎", "🍌", "🍇"

함수 선언

function sayHi(name) {
return `hello, ${name}`;
}

console.log(sayHi("js"));
hello, js

문자열 템플릿: ${"{값}"}

let skill = "js";
console.log(`내가 배우는 기술은 ${skill}야!`);
내가 배우는 기술은 js야!
