DartPad 문법 요약 카드

final, const

  • final : 코드를 실행할 때 값이 확정
  • const : 코드 실행 전에 이미 확정되어야 함
void main() {   // void: 아무 값도 반환하지 않음
  print('Hello World');
}
Hello World

void main() {
  var name = 'dart';
  print(name);
}
dart

void main() {
  var name = 'dart-pad';
  print(name);
}
dart-pad

void main() {
  final DateTime now = DateTime.now();    // const 사용 시 에러 발생
    print(now);
}
2025-07-18 10:25:43.123456

var, String, int, double, bool

  • var : 타입을 자동 추론, 명시해주면 더 명확하고 유지보수에 좋음
  • String : 문자열
  • int : 정수
  • double : 실수
  • bool : true 또는 false (논리값)
void main() {
  String name = 'string';

  print(name);
}
string

void main() {
   int isInt = 10;

  print(isInt);
}
10

void main() {
  double isDouble = 2.5;

  print(name);
}
2.5

void main() {
  bool isTrue = true;

  print(name);
}
true
